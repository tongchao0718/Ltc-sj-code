## 核心设计理念

### 离线优先，本地用不了再用云

**设计原则**：

1. **数据本地存储优先**：所有数据默认存储在本地浏览器 IndexedDB/SQLite
2. **云端同步可选**：仅在有云端数据库时才启用同步功能
3. **离线完整可用**：即使没有网络，核心功能必须完整可用
4. **渐进式增强**：有云端时享受协作功能，无云端时享受离线便利

### 架构对比

| 场景     | 传统方案  | 本方案（离线优先）  |
| ------ | ----- | ---------- |
| 有云端数据库 | ✅ 支持  | ✅ 支持（完整功能） |
| 无云端数据库 | ❌ 不支持 | ✅ 支持（离线模式） |
| 网络不稳定  | ❌ 体验差 | ✅ 离线可用     |
| 数据同步   | 必须联网  | 可选同步       |

### 1.2 本地数据库技术选型

#### 方案一：IndexedDB（推荐 - 无需安装）

**特点**：

- 浏览器内置，无需安装任何软件
- 容量大（通常50MB-无限制）
- 支持复杂查询
- 数据持久化存储

**适用场景**：

- 纯离线使用
- 网络不稳定环境
- 临时测试环境

**示例代码**：

```javascript
// 使用 Dexie.js 封装 IndexedDB

  projects: '++id, projectCode, projectName, ownerId, createTime',
  assets: '++id, assetCode, assetName, assetType, projectId',
  settings: 'key, value',
  syncQueue: '++id, tableName, operation, data, timestamp'

// 使用示例
await db.projects.add({
  projectCode: 'PROJ001',
  projectName: '我的项目',
  ownerId: 1,
  createTime: new Date()
```

#### 方案二：SQLite（桌面应用场景）

**特点**：

- 单文件数据库，部署简单
- 性能优秀，支持SQL查询
- 跨平台支持

**适用场景**：

- Electron桌面应用
- 本地长期使用
- 数据量较大时

**示例代码**：


### 1.3 数据同步策略

#### 冲突解决机制

```javascript
// 冲突解决策略：本地优先 + 云端备份
class SyncManager {
  async sync() {
    if (!this.isOnline() || !this.hasCloudDB()) {
      console.log('离线模式：数据存储在本地');
      return;
    }

    // 1. 推送本地变更到云端
    const pendingChanges = await this.getPendingChanges();
    for (const change of pendingChanges) {
      try {
        await this.pushToCloud(change);
        await this.markAsSynced(change.id);
      } catch (error) {
        if (error.type === 'CONFLICT') {
          // 冲突：使用本地版本覆盖（默认策略）
          await this.resolveConflict(change, 'LOCAL_WINS');
        }
      }
    }

    // 2. 拉取云端变更到本地
    const cloudChanges = await this.fetchCloudChanges();
    for (const change of cloudChanges) {
      if (!this.isLocallyModified(change)) {
        await this.pullToLocal(change);
      }
    }
  }

  // 冲突解决策略
  resolveConflict(local, remote, strategy = 'LOCAL_WINS') {
    switch (strategy) {
      case 'LOCAL_WINS':
        return local; // 本地优先（默认）
      case 'REMOTE_WINS':
        return remote; // 云端优先
      case 'MERGE':
        return this.mergeData(local, remote); // 合并
      case 'ASK_USER':
        return this.askUserToResolve(local, remote); // 用户选择
    }
  }
}
```

## 2. 离线优先的技术实现

### 2.1 前端架构（离线优先）

```javascript
// main.js - 应用入口
import { createApp } from 'vue';
import App from './App.vue';
import router from './router';
import store from './store';
import { initDatabase } from './database';
import { syncManager } from './sync';

// 检测存储模式
const storageMode = detectStorageMode();
// 'local-only': 仅本地存储
// 'cloud-sync': 本地 + 云端同步

async function bootstrap() {
  // 1. 初始化本地数据库
  await initDatabase();

  // 2. 创建Vue应用
  const app = createApp(App);

  // 3. 根据存储模式配置API服务
  if (storageMode === 'cloud-sync') {
    // 云端同步模式
    store.dispatch('user/login');
    await syncManager.startSync();
  } else {
    // 纯离线模式
    console.log('运行在纯离线模式');
  }

  app.use(store);
  app.use(router);
  app.mount('#app');
}

bootstrap();

// 检测存储模式函数
function detectStorageMode() {
  // 优先使用环境变量配置
  if (process.env.VUE_APP_STORAGE_MODE) {
    return process.env.VUE_APP_STORAGE_MODE;
  }

  // 检测是否配置了云端API地址
  if (process.env.VUE_APP_API_BASE_URL) {
    return 'cloud-sync';
  }

  // 默认纯离线模式
  return 'local-only';
}
```

### 2.2 API服务封装（离线兼容）

```javascript
// api/ProjectService.js
import { db } from '../database/local';
import { cloudApi } from './cloud';

class ProjectService {
  constructor() {
    this.useCloud = this.checkCloudAvailable();
  }

  checkCloudAvailable() {
    return process.env.VUE_APP_API_BASE_URL && navigator.onLine;
  }

  // 创建项目 - 离线优先
  async createProject(projectData) {
    const project = {
      ...projectData,
      id: this.generateId(),
      _syncStatus: this.useCloud ? 'pending' : 'synced',
      createTime: new Date().toISOString()
    };

    // 1. 先保存到本地
    await db.projects.add(project);

    // 2. 如果有云端且在线，推送到云端
    if (this.useCloud && navigator.onLine) {
      try {
        await cloudApi.projects.create(project);
        await this.updateSyncStatus(project.id, 'synced');
      } catch (error) {
        console.warn('云端同步失败，数据已保存在本地:', error);
      }
    }

    return project;
  }

  // 获取项目列表 - 智能路由
  async getProjects() {
    if (this.useCloud && navigator.onLine) {
      try {
        // 在线：获取云端数据并同步到本地
        const cloudData = await cloudApi.projects.list();
        await this.syncFromCloud(cloudData);
        return cloudData;
      } catch (error) {
        console.warn('云端获取失败，使用本地数据:', error);
        return await db.projects.toArray();
      }
    } else {
      // 离线：使用本地数据
      return await db.projects.toArray();
    }
  }

  // 同步本地数据到云端
  async syncToCloud() {
    if (!this.useCloud || !navigator.onLine) {
      return { success: false, reason: 'offline' };
    }

    const pendingItems = await db.projects
      .where('_syncStatus')
      .equals('pending')
      .toArray();

    for (const item of pendingItems) {
      try {
        await cloudApi.projects.update(item.id, item);
        await this.updateSyncStatus(item.id, 'synced');
      } catch (error) {
        console.error('同步失败:', error);
      }
    }

    return { success: true, synced: pendingItems.length };
  }

  // 更新同步状态
  async updateSyncStatus(id, status) {
    await db.projects.update(id, { _syncStatus: status });
  }
}

export const projectService = new ProjectService();
```

### 2.3 离线检测与自动切换

```javascript
// utils/NetworkStatus.js
import { ref, computed } from 'vue';

const isOnline = ref(navigator.onLine);
const wasOffline = ref(false); // 记录是否曾经离线

// 监听网络状态变化
window.addEventListener('online', () => {
  isOnline.value = true;

  if (wasOffline.value) {
    // 从离线恢复，触发同步
    console.log('网络恢复，触发数据同步');
    EventBus.emit('network:restored');
  }
});

window.addEventListener('offline', () => {
  isOnline.value = false;
  wasOffline.value = true;
  console.log('网络断开，切换到离线模式');
  EventBus.emit('network:lost');
});

// 导出组合式函数
export function useNetworkStatus() {
  const networkStatus = computed(() => {
    if (isOnline.value) return 'online';
    return 'offline';
  });

  return {
    isOnline: readonly(isOnline),
    networkStatus,
    onNetworkRestored: (callback) => EventBus.on('network:restored', callback),
    onNetworkLost: (callback) => EventBus.on('network:lost', callback)
  };
}
```

| 方案                   | 数据库     | 适用场景   | 优点        |

| Node.js + SQLite     | SQLite  | 轻量级需求  | 开发快，资源占用少 |

### 10.3 打包配置详解

#### 10.3.1 Vite 配置文件

```javascript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import { resolve } from 'path'
import fs from 'fs'

// 自定义插件：在构建完成后复制和重命名 HTML 文件
const copyHtmlPlugin = () => ({
  name: 'copy-html',
  closeBundle() {
    const distPath = resolve(__dirname, '../dist/appX')
    const buildPath = resolve(distPath, 'build')
    const buildSrcPath = resolve(__dirname)
    
    // 如果 build 目录存在，移动文件到根目录
    if (fs.existsSync(buildPath)) {
      // 读取生成的 appX.html 文件
      let htmlContent = fs.readFileSync(resolve(buildPath, 'appX.html'), 'utf8')
      
      // 修正资源引用路径，将 '../assets/' 改为 './assets/'
      htmlContent = htmlContent.replace(/\.\.\/assets\//g, './assets/')
      
      // 写入修正后的 appX.html
      fs.writeFileSync(resolve(distPath, 'appX.html'), htmlContent)
      
      // 复制为 index.html
      fs.writeFileSync(resolve(distPath, 'index.html'), htmlContent)
      
      // 删除 build 目录
      fs.rmSync(buildPath, { recursive: true })
    }
    
    // 复制 start.html (CDN版本) 到根目录
    if (fs.existsSync(resolve(buildSrcPath, 'start.html'))) {
      let startHtml = fs.readFileSync(resolve(buildSrcPath, 'start.html'), 'utf8')
      // 替换占位符为实际的JS文件名
      const jsFiles = fs.readdirSync(resolve(distPath, 'assets')).filter(file => file.endsWith('.js'))
      if (jsFiles.length > 0) {
        const mainJsFile = jsFiles.find(file => file.includes('main')) || jsFiles[0]
        startHtml = startHtml.replace('__MAIN_JS__', `./assets/${mainJsFile}`)
      }
      fs.writeFileSync(resolve(distPath, 'start.html'), startHtml)
    }
  }
})

export default defineConfig({
  plugins: [vue(), copyHtmlPlugin()],
  base: './', // 使用相对路径
  build: {
    outDir: resolve(__dirname, '../dist/appX'),
    assetsDir: 'assets',
    minify: 'terser',
    terserOptions: {
      compress: {
        drop_console: true
      }
    },
    rollupOptions: {
      output: {
        format: 'iife', // 使用 IIFE 格式，避免模块加载问题
        entryFileNames: 'assets/[name]-[hash].js',
        chunkFileNames: 'assets/[name]-[hash].js',
        assetFileNames: 'assets/[name]-[hash].[ext]'
      }
    }
  },
  resolve: {
    alias: {
      '@': resolve(__dirname, '../src')
    }
  }
})
```

- **API 服务**：子应用应实现离线优先的 API 服务封装
- **网络状态**：子应用应监听网络状态变化，自动切换在线/离线模式
- **数据同步**：子应用应支持数据同步功能，确保多设备数据一致性
