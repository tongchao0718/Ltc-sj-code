# 一键更新指南

## 一、概述

### 1. 什么是一键更新

一键更新是指通过简单的命令或操作，自动完成系统版本更新的全过程，包括检查更新、下载更新包、安装更新、验证更新等步骤，无需用户手动干预。

### 2. 核心优势

- **操作简单**：只需一条命令或一次点击
- **自动化**：全过程自动执行，减少人工操作
- **安全可靠**：内置备份和回滚机制
- **增量更新**：只更新必要文件，减少下载量
- **用户友好**：清晰的进度提示和错误处理

### 3. 适用场景

- **开发团队内部更新**：快速同步代码
- **生产环境部署**：自动化部署流程
- **终端用户更新**：简化用户更新体验
- **多环境同步**：确保各环境版本一致

### 1. 命令行一键更新

#### 1.1 脚本实现

**创建 `scripts/update.js`**：

```javascript
#!/usr/bin/env node
/**
 * 一键更新脚本
 * Usage: npm run update
 */

const fs = require('fs');
const path = require('path');
const { execSync } = require('child_process');

class UpdateManager {
  constructor() {
    this.currentVersion = this.getCurrentVersion();
    this.backupDir = `backups/${Date.now()}`;
  }

  // 获取当前版本
  getCurrentVersion() {
    try {
      const packageJson = require('../package.json');
      return packageJson.version;
    } catch (error) {
      console.error('读取版本号失败:', error);
      return '0.0.0';
    }
  }

  // 检查更新
  async checkUpdate() {
    console.log('🔍 检查更新中...');
    
    try {
      // 模拟检查更新（实际项目中应调用更新服务器API）
      await new Promise(resolve => setTimeout(resolve, 1000));
      
      return {
        latestVersion: '1.1.0',
        currentVersion: this.currentVersion,
        updateAvailable: true,
        changelog: [
          '✨ 新增数据持久化功能',
          '🐛 修复数据导出bug',
          '⚡ 优化性能'
        ]
      };
    } catch (error) {
      console.error('检查更新失败:', error);
      return { updateAvailable: false };
    }
  }

  // 执行更新
  async update() {
    const updateInfo = await this.checkUpdate();
    
    if (!updateInfo.updateAvailable) {
      console.log('✅ 当前已是最新版本');
      return;
    }

    console.log(`📦 发现新版本 v${updateInfo.latestVersion}`);
    console.log('📝 更新内容:');
    updateInfo.changelog.forEach(item => console.log(`  ${item}`));

    // 确认更新
    const readline = require('readline').createInterface({
      input: process.stdin,
      output: process.stdout
    });

    return new Promise((resolve) => {
      readline.question('是否继续更新? (y/n): ', (answer) => {
        readline.close();
        
        if (answer.toLowerCase() !== 'y') {
          console.log('取消更新');
          resolve(false);
          return;
        }

        this.doUpdate(updateInfo).then(resolve);
      });
    });
  }

  // 执行更新操作
  async doUpdate(updateInfo) {
    try {
      console.log('🚀 开始更新...');
      
      // 1. 备份当前版本
      this.backupCurrentVersion();
      
      // 2. 拉取最新代码
      this.pullLatestCode();
      
      // 3. 安装依赖
      this.installDependencies();
      
      // 4. 构建项目
      this.buildProject();
      
      // 5. 验证更新
      this.verifyUpdate();
      
      console.log('✅ 更新成功！');
      console.log(`🎉 已更新到 v${updateInfo.latestVersion}`);
      return true;
    } catch (error) {
      console.error('更新失败:', error);
      this.rollback();
      return false;
    }
  }

  // 备份当前版本
  backupCurrentVersion() {
    console.log('📁 备份当前版本...');
    execSync(`mkdir -p ${this.backupDir}`);
    execSync(`robocopy . ${this.backupDir} /E /XD node_modules backups dist`);
  }

  // 拉取最新代码
  pullLatestCode() {
    console.log('📡 拉取最新代码...');
    execSync('git pull origin main');
  }

  // 安装依赖
  installDependencies() {
    console.log('📦 安装依赖...');
    execSync('npm install');
  }

  // 构建项目
  buildProject() {
    console.log('🔨 构建项目...');
    execSync('npm run build');
  }

  // 验证更新
  verifyUpdate() {
    console.log('✅ 验证更新...');
    // 这里可以添加验证逻辑
  }

  // 回滚
  rollback() {
    console.log('🔄 回滚到之前版本...');
    // 回滚逻辑
  }
}

// 执行更新
const updater = new UpdateManager();
updater.update().then(success => {
  if (success) {
    console.log('📢 更新完成，请重启应用');
  }
  process.exit(success ? 0 : 1);
});
```

```javascript
// 跨平台命令处理
class CrossPlatform {
  // 获取平台特定命令
  static getCommand(command) {
    const isWindows = process.platform === 'win32';
    
    const commands = {
      mkdir: isWindows ? 'mkdir' : 'mkdir -p',
      copy: isWindows ? 'robocopy' : 'cp -r',
      remove: isWindows ? 'rmdir /s /q' : 'rm -rf'
    };
    
    return commands[command] || command;
  }

  // 执行跨平台命令
  static exec(command, options = {}) {
    const platformCommand = this.getCommand(command);
    execSync(platformCommand, options);
  }
}
```

    latestVersion: '1.1.0',
    currentVersion: req.query.version || '1.0.0',
    updateAvailable: true,
    changelog: [
      '✨ 新增数据持久化功能',
      '🐛 修复数据导出bug',
      '⚡ 优化性能'
    ],
    downloadUrl: 'https://api.example.com/update/1.1.0'
  });
});

// 更新包下载
app.get('/update/:version', (req, res) => {
  const version = req.params.version;
  // 提供更新包下载
  res.download(`updates/${version}.zip`);
});

app.listen(3000, () => {
  console.log('更新服务器运行在 http://localhost:3000');
});
